\newglossaryentry{torque}{

\newglossaryentry{stalltorqure}{

}

\newglossaryentry{drumbrake}{
	name={drum brake},
	description={Type of brake that uses friction pads to stop rotation of wheels}

\newglossaryentry{axle}{
	name={axle},
	description={Shaft that connects and/or drives two wheels}
}

\newglossaryentry{conical}{
	name={conical wheels},
	description={The shape of wheels commonly used for locomotive wheels which allow turning around corners}
}
\newglossaryentry{gearbox}{
	name={gearbox},

}
